  1. npm i react-router-dom

# install Tailwind css 
1. cmd = npm install tailwindcss @tailwindcss/vite

2. vite.config.js

   import { defineConfig } from 'vite'
    import tailwindcss from '@tailwindcss/vite'
    
    export default defineConfig({
      plugins: [
        tailwindcss(),
      ],
    })

3. index.css
      @import "tailwindcss";
